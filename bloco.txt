    }

    sleep(2);

        out = argv[2];
    }
    else if (argc == 5 && strcmp(argv[1], "-t") == 0 && strcmp(argv[3], "-s") == 0) {
        time = atoi(argv[2]);
        out = argv[4];
    }
    else {
        printf("Uso:\n");
        printf("%s -s saida\n", argv[0]);
        printf("%s -t tempo -s saida\n", argv[0]);
        return 1;
    }

    if(strcmp(out, "on") == 0) {
        fprintf(serial,"1\n");
    }
    else if(strcmp(out, "off") == 0) {
        fprintf(serial,"0\n");
    }
    else {
        printf("Saida desconhecida\n");
    }

    fclose(serial);
    return 0;
}
